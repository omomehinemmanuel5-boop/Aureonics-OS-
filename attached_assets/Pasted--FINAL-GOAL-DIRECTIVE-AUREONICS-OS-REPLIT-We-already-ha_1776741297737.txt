## 3.2 Basin Force (simplex-consistent)

```python
def compute_basin_force(x, input_data):
    eps = 1e-4
    grad = []

    for i in range(len(x)):
        x_up = x.copy()
        x_down = x.copy()
        x_up[i] += eps
        x_down[i] -= eps

        phi_up = compute_phi(x_up, input_data)
        phi_down = compute_phi(x_down, input_data)

        dphi = (phi_up - phi_down) / (2 * eps)
        grad.append(dphi)

    mean_grad = sum(grad) / len(grad)
    lambda_gain = 0.2

    force = [-(g - mean_grad) * lambda_gain for g in grad]
    return cap_force(force)
```

# ⚖️ 4. FORCE CAPPING

```python
def cap_force(v, max_norm=1.0):
    norm = sum(abs(x) for x in v)
    if norm > max_norm and norm > 0:
        scale = max_norm / norm
        return [x * scale for x in v]
    return v
```
